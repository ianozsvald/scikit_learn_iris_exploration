def plot_feature(feature_nbr):
    # plot the iris data
    f=pylab.plot(iris.data[:,feature_nbr])[0]
    # make a second axis sharing the x-axis, separate the 3 classes with vertical lines
    ax2=f.axes.twinx()
    ax2.vlines(50, 0, 1)
    ax2.vlines(100, 0, 1)
    f.axes.set_title(iris.feature_names[feature_nbr])
    return f
